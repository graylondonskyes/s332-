function normalizeSectionsData(raw){
  if(!raw) return null;
  if(raw.categories && Array.isArray(raw.categories)){
    const out = {};
    for(const c of raw.categories){
      if(!c || !c.name || !Array.isArray(c.items)) continue;
      out[c.name] = c.items.filter(x=>x && x.html).map(x=>({
        id: String(x.id||("sec_"+Math.random().toString(16).slice(2))),
        title: String(x.title||x.id||"Section"),
        html: String(x.html)
      }));
    }
    return out;
  }
  const out = {};
  for(const k of Object.keys(raw)){
    if(Array.isArray(raw[k])){
      out[k] = raw[k].filter(x=>x && x.html).map(x=>({
        id: String(x.id||("sec_"+Math.random().toString(16).slice(2))),
        title: String(x.title||x.id||"Section"),
        html: String(x.html)
      }));
    }
  }
  return out;
}

/* ---------- SECTIONS UI ---------- */
const catSelect = qs("#catSelect");
const secSelect = qs("#secSelect");
const homeStackPre = qs("#homeStack");

function populateCategories(){
  catSelect.innerHTML = "";
  const cats = Object.keys(state.sectionsData||{});
  for(const c of cats){
    const opt = document.createElement("option");
    opt.value = c; opt.textContent = c;
    catSelect.appendChild(opt);
  }
  if(cats.length) catSelect.value = cats[0];
  populateSections();
}
function populateSections(){
  secSelect.innerHTML = "";
  const cat = catSelect.value;
  const items = (state.sectionsData && state.sectionsData[cat]) ? state.sectionsData[cat] : [];
  for(const it of items){
    const opt = document.createElement("option");
    opt.value = it.id;
    opt.textContent = it.title;
    secSelect.appendChild(opt);
  }
}
catSelect.addEventListener("change", populateSections);

function renderHomeStack(){
  if(!state.homeStack.length){
    homeStackPre.textContent = "(no sections selected)\n\nAdd sections to pre-build your Home page.";
    return;
  }
  homeStackPre.textContent = state.homeStack.map((s,i)=>{
    const mark = (i===state.selectedIndex) ? "▶" : " ";
    return `${mark} [${i+1}] ${s.title} (${s.id})`;
  }).join("\n");
}

qs("#btnAddSection").addEventListener("click", ()=>{
  const cat = catSelect.value;
  const id = secSelect.value;
  const items = (state.sectionsData && state.sectionsData[cat]) ? state.sectionsData[cat] : [];
  const found = items.find(x=>x.id === id);
  if(!found){ showToast("Pick a section"); return; }
  state.homeStack.push(found);
  state.selectedIndex = state.homeStack.length - 1;
  renderHomeStack(); updatePreview(); showToast("Added");
});

qs("#btnClearSections").addEventListener("click", ()=>{
  state.homeStack = []; state.selectedIndex = -1;
  renderHomeStack(); updatePreview(); showToast("Cleared");
});

qs("#btnMoveUp").addEventListener("click", ()=>{
  const i = state.selectedIndex; if(i<=0) return;
  [state.homeStack[i-1], state.homeStack[i]] = [state.homeStack[i], state.homeStack[i-1]];
  state.selectedIndex = i-1; renderHomeStack(); updatePreview(); showToast("Moved");
});

qs("#btnMoveDown").addEventListener("click", ()=>{
  const i = state.selectedIndex; if(i<0 || i>=state.homeStack.length-1) return;
  [state.homeStack[i+1], state.homeStack[i]] = [state.homeStack[i], state.homeStack[i+1]];
  state.selectedIndex = i+1; renderHomeStack(); updatePreview(); showToast("Moved");
});

qs("#btnRemove").addEventListener("click", ()=>{
  const i = state.selectedIndex; if(i<0) return;
  state.homeStack.splice(i,1);
  state.selectedIndex = Math.min(i, state.homeStack.length-1);
  renderHomeStack(); updatePreview(); showToast("Removed");
});

/* ---------- LOAD SECTIONS JSON ---------- */
qs("#btnLoadSections").addEventListener("click", async ()=>{
  const inp = document.createElement("input");
  inp.type = "file";
  inp.accept = "application/json,.json";
  inp.onchange = async ()=>{
    const file = inp.files && inp.files[0];
    if(!file) return;
    try{
      const raw = JSON.parse(await file.text());
      const norm = normalizeSectionsData(raw);
      if(!norm || !Object.keys(norm).length) throw new Error("No sections found");
      state.sectionsData = norm;
      qs("#sectionsStatus").className = "ok";
      qs("#sectionsStatus").textContent = "Custom sections JSON loaded.";
      state.homeStack = []; state.selectedIndex = -1;
      populateCategories(); renderHomeStack(); updatePreview();
      showToast("Library loaded");
    }catch(e){
      qs("#sectionsStatus").className = "warn";
      qs("#sectionsStatus").textContent = "Failed to load JSON. Format: {\"Category\":[{\"id\",\"title\",\"html\"}...]}";
      showToast("Load failed");
    }
  };
  inp.click();
});

/* ---------- BUILD ZIP REPO ---------- */
async function buildRepoZip(cfg){
  if(typeof window.JSZip !== "function"){
    throw new Error("JSZip not available (CDN blocked).");
  }

  const zip = new JSZip();
  const root = zip.folder("public-clean");
  const assets = root.folder("assets");
  const css = assets.folder("css");
  const js = assets.folder("js");

  css.file("tokens.css", buildTokensCss(cfg));
  css.file("base.css", BASE_CSS);
  css.file("components.css", COMPONENTS_CSS);

  js.file("license.config.js", buildLicenseJs(cfg.tier));
  js.file("nav.config.js", buildNavJs(cfg));
  js.file("footer.config.js", buildFooterJs(cfg));
  js.file("app.js", APP_JS);

  root.file("index.html", buildHomeHtml(cfg));
  if(cfg.pages.includes("pricing")) root.folder("pricing").file("index.html", buildPricingHtml(cfg));
  if(cfg.pages.includes("sectionsHub")) root.folder("sections").file("index.html", buildSectionsHubHtml(cfg));
  if(cfg.pages.includes("about")) root.folder("about").file("index.html", buildSimplePageHtml(cfg, "About", "Tell your story. Paste sections below."));
  if(cfg.pages.includes("contact")) root.folder("contact").file("index.html", buildSimplePageHtml(cfg, "Contact", "Add your form embed or contact details below."));

  zip.file("README_START_HERE.txt", README_TXT);
  zip.file("CLIENT_TUTORIAL.txt", CLIENT_TUTORIAL_TXT);
  zip.file("LICENSE_SUMMARY.txt", LICENSE_SUMMARY_TXT);
  zip.file("TERMS_LICENSE.txt", TERMS_LICENSE_TXT);

  zip.file("STUDIO_BUILD_MANIFEST.json", JSON.stringify({
    builtAt: new Date().toISOString(),
    brandName: cfg.brandName,
    tier: cfg.tier,
    pages: cfg.pages,
    homeSections: cfg.homeSections.map(s=>({id:s.id,title:s.title}))
  }, null, 2));

  return zip;
}

/* ---------- GENERATE ---------- */
qs("#btnGenerate").addEventListener("click", async ()=>{
  try{
    setMini("Generating ZIP…");
    const cfg = collectConfig();
    const zip = await buildRepoZip(cfg);
    const blob = await zip.generateAsync({type:"blob"});
    const a = document.createElement("a");
    a.href = URL.createObjectURL(blob);
    a.download = `RepoKitStudio_${safeSlug(cfg.brandName)}_${cfg.tier}.zip`;
    a.click();
    URL.revokeObjectURL(a.href);
    setMini("ZIP generated and downloaded.");
    showToast("ZIP downloaded");
  }catch(err){
    console.error(err);
    setMini("ZIP failed. Open console for error details.");
    showToast("ZIP failed");
  }
});

function safeSlug(s){
  return String(s||"site").toLowerCase().trim().replace(/[^a-z0-9]+/g,"-").replace(/^-+|-+$/g,"").slice(0,50) || "site";
}
function escapeHtml(s){
  return String(s).replaceAll("&","&").replaceAll("<","<").replaceAll(">",">").replaceAll('"',""");
}
function escapeAttr(s){ return escapeHtml(s).replaceAll("'","'"); }
